using UnityEngine.Events;

public enum QuesCallBacksName
{
点击关闭,
移入选项1,
移入选项2,
移入选项3,
移入选项4,
}

[System.Serializable]
public class QuesItem
{
public string tag;
public string ques;
public List options;
public string answer;
}
[System.Serializable]
public class Root
{
public List database = new List();
}
public class QuestionsControl_json : MonoSingleton<QuestionsControl_json>
{
public GameObject togglePrefab;
public Sprite rightImg;
public Sprite wrongImg;

private Text ui_Ques;
private Button ok;
private Button close;
private ToggleGroup toggleGroup;
private QuesItem currentData;
private Root r;
//private bool resoult;
//private bool single;//是单选
private string filePath;
[HideInInspector]
public UnityEvent CloseCallBack, Option0CallBack, Option1CallBack, Option2CallBack, Option3CallBack;

​ 

void Awake()
{
    filePath = Application.dataPath + "/StreamingAssets/Questions.json";
    StreamReader streamreader = new StreamReader(filePath, Encoding.UTF8);//读取数据，转换成数据流
    JsonReader js = new JsonReader(streamreader);//再转换成json数据
    r = JsonMapper.ToObject<Root>(js);//读取

    ui_Ques = Global.FindChild<Text>(transform, "Ques");
    ok = Global.FindChild<Button>(transform, "确定Button");
    close = Global.FindChild<Button>(transform, "关闭Button");
    ok.onClick.AddListener(delegate { Judge(); });
    close.onClick.AddListener(ClosePanel);
    toggleGroup = Global.FindChild<ToggleGroup>(transform, "ToggleGroup");
}


/// <summary>
/// 往Json文件中存数据
/// </summary>
/// <param name="ques">QuesItem对象</param>
public void SaveQues(QuesItem ques)
{
    bool isAdd = true;//没有tag(isAdd为true)则添加
    foreach (QuesItem q in r.database)
    {
        if (q.tag == ques.tag)//查找符合参数的题目
        {
            isAdd = false;
        }
    }

    if (isAdd)
    {
        r.database.Add(ques);
    }
    else
    {
        Debug.Log("Json表中已有此tag值");
    }

    string jsonStr = JsonMapper.ToJson(r);//将ques对象转化为Json字符串
    StreamWriter writer = new StreamWriter(filePath);
    writer.Write(jsonStr);//将字符串写入到文件中
    writer.Close();
    Debug.Log("SAVE");
}

/// <summary>
/// 添加回调
/// </summary>
/// <param name="name">要添加回调的状态</param>
/// <param name="unityAction">添加的方法</param>
public void AddCallBacks(QuesCallBacksName name,UnityAction unityAction)
{
    switch (name)
    {
        case QuesCallBacksName.点击关闭:
            if (CloseCallBack != null)
                CloseCallBack.RemoveAllListeners();
            CloseCallBack.AddListener(unityAction);
            break;
        case QuesCallBacksName.移入选项1:
            if (Option0CallBack != null)
                Option0CallBack.RemoveAllListeners();
            Option0CallBack.AddListener(unityAction);
            break;
        case QuesCallBacksName.移入选项2:
            if (Option1CallBack != null)
                Option1CallBack.RemoveAllListeners();
            Option1CallBack.AddListener(unityAction);
            break;
        case QuesCallBacksName.移入选项3:
            if (Option2CallBack != null)
                Option2CallBack.RemoveAllListeners();
            Option2CallBack.AddListener(unityAction);
            break;
        case QuesCallBacksName.移入选项4:
            if (Option3CallBack != null)
                Option3CallBack.RemoveAllListeners();
            Option3CallBack.AddListener(unityAction);
            break;
    }
}

/// <summary>
/// 在题库中查找标签出题
/// </summary>
/// <param name="tag">标签</param>
public void QuesShow(string tag)
{
    if (toggleGroup.transform.childCount > 0)//清空之前的选项
    {
        for (int i = 0; i < toggleGroup.transform.childCount; i++)
        {
            Destroy(toggleGroup.transform.GetChild(i).gameObject);
        }
    }
    close.gameObject.SetActive(false);
    foreach (QuesItem q in r.database)
    {
        if (q.tag == tag)//查找符合参数的题目
        {
            currentData = q;

            if (currentData.options.Count > 0)//选项不为空则实例化选项
            {
                for (int i = 0; i < currentData.options.Count; i++)
                {
                    GameObject temp = GameObject.Instantiate(togglePrefab);
                    temp.transform.SetParent(toggleGroup.transform);
                    temp.transform.localScale = Vector3.one;
                    Global.FindChild<Text>(temp.transform, "Label").text = currentData.options[i];
                    if (currentData.answer.Length == 1)//答案长度为1为单选，否则为多选
                    {
                        temp.GetComponent<Toggle>().group = toggleGroup;
                        ok.gameObject.SetActive(false);
                        ui_Ques.text = currentData.ques+"           (单选) ";
                        temp.GetComponent<Toggle>().onValueChanged.AddListener(ifselect => { if (ifselect) Judge(); });
                }
                    else {
                        ok.gameObject.SetActive(true);
                        ui_Ques.text = currentData.ques + "           (多选) ";
                    }
                }
            }
            break;
        }
    }
    Global.FindChild<Transform>(transform, "QesPanel").DOLocalMove(new Vector3(-30, -320, 0), 1.5f).ChangeStartValue(new Vector3(-1900, -320, 0)).SetEase(Ease.InOutBack);
}

private void Judge()
{
    List<GameObject> chooses = new List<GameObject>();
    string options = "";
    foreach (Transform t in toggleGroup.transform)
    {
        t.GetComponent<Toggle>().onValueChanged.RemoveAllListeners();
        if (t.GetComponent<Toggle>().isOn)
        {
            options += t.GetComponentInChildren<Text>().text.Substring(0, 1);
            chooses.Add(t.gameObject);
        } 
    }
    if (options == "")
    {
        transform.Find("QesPanel/提示").gameObject.SetActive(true);
    }
    else
    {
        transform.Find("QesPanel/提示").gameObject.SetActive(false);
        if (options.Trim() == currentData.answer)//答案正确
        {
            AudioMan.Instance.PlaySound(SoundType.正确提示);
            for (int i = 0; i < chooses.Count; i++)
            {
                chooses[i].GetComponentInChildren<Text>().color = new Color(0.1f, 1f, 0.1f, 1);
            }
            foreach (Transform t in toggleGroup.transform)
            {
                t.GetComponent<Toggle>().interactable = false;
            }
        }
        else//答案错误
        {
            AudioMan.Instance.PlaySound(SoundType.错误提示);
            foreach (Transform t in toggleGroup.transform)//错误答案变红
            {
                t.GetComponent<Toggle>().interactable = false;
                string s = t.GetComponentInChildren<Text>().text.Substring(0, 1);
                if (options.Trim().Contains(s))
                {
                    t.GetComponentInChildren<Text>().color = new Color(1f, 0.1f, 0.1f, 1);
                    t.GetComponent<Toggle>().group = null;
                    t.GetComponent<Toggle>().isOn = true;
                    t.Find("Checkmark").GetComponent<Image>().sprite = wrongImg;
                }
            }
            foreach (Transform t in toggleGroup.transform)//正确答案变绿
            {
                string s = t.GetComponentInChildren<Text>().text.Substring(0, 1);
                if (currentData.answer.Contains(s))
                {
                    t.GetComponentInChildren<Text>().color = new Color(0.1f, 1f, 0.1f, 1);
                    t.GetComponent<Toggle>().group = null;
                    t.GetComponent<Toggle>().isOn = true;
                    t.Find("Checkmark").GetComponent<Image>().sprite = rightImg;
                }
            }
        }
        ok.gameObject.SetActive(false);
        close.gameObject.SetActive(true);
    }
}

public void  ClosePanel()
{
    Global.FindChild<Transform>(transform, "QesPanel").DOLocalMove(new Vector3(-1900, -320, 0), 1.5f).ChangeStartValue(new Vector3(-30, -320, 0)).SetEase(Ease.InOutBack);
    if (CloseCallBack != null)
    {
        CloseCallBack.Invoke();
    }
}

/// <summary>
/// 初始化
/// </summary>
public void Init()
{
    Global.FindChild<Transform>(transform, "QesPanel").transform.position=new Vector3(-1900, -320, 0);
    CloseCallBack.RemoveAllListeners();
}

}

//每个题最多四个选项，鼠标移入移出事件

using DG.Tweening;
using UnityEngine;
using UnityEngine.EventSystems;

public class OptionEffect : MonoBehaviour,IPointerEnterHandler,IPointerExitHandler{

public void OnPointerEnter(PointerEventData eventData)
{
    transform.Find("Background").DOLocalRotate(new Vector3(0, 0, 45), 0.1f).ChangeStartValue(Vector3.zero).SetEase(Ease.Linear);
    switch (transform.GetSiblingIndex())
    {
        case 0:
            if (QuestionsControl_json.Instance.Option0CallBack != null)
            {
                QuestionsControl_json.Instance.Option0CallBack.Invoke();
                QuestionsControl_json.Instance.Option0CallBack.RemoveAllListeners();
            }
            break;
        case 1:
            if (QuestionsControl_json.Instance.Option1CallBack != null)
            {
                QuestionsControl_json.Instance.Option1CallBack.Invoke();
                QuestionsControl_json.Instance.Option1CallBack.RemoveAllListeners();
            }
            break;
        case 2:
            if (QuestionsControl_json.Instance.Option2CallBack != null)
            {
                QuestionsControl_json.Instance.Option2CallBack.Invoke();
                QuestionsControl_json.Instance.Option2CallBack.RemoveAllListeners();
            }
            break;
        case 3:
            if (QuestionsControl_json.Instance.Option3CallBack != null)
            {
                QuestionsControl_json.Instance.Option3CallBack.Invoke();
                QuestionsControl_json.Instance.Option3CallBack.RemoveAllListeners();
            }
            break;
    }
}

public void OnPointerExit(PointerEventData eventData)
{
    transform.Find("Background").DOLocalRotate(Vector3.zero, 0.1f).ChangeStartValue(new Vector3(0, 0, 45)).SetEase(Ease.Linear);
}

}
